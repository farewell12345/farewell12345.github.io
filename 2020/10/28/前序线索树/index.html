前序线索树

Posted on 2020-10-28- Edited on 2020-11-06
1.2k- 5m-
数据结构

在线索化二叉树中，被线索化的节点，空闲的右指针指向后继节点，空闲的左指针指向前驱节点

typedef struct node{
   int data; // 数据域
   struct node* left,*right; // 指针域
   int leftTag,rightTag; // 标记左右指针是否为线索
}node;



/*创建二叉树*/
node *createTree(node *head){
   if(head == NULL){
     int num = 0;
     scanf("%d",&num);
     if(num != -1){
       head =(node*) malloc(sizeof(node));
       head->right =NULL;
       head->left = NULL;
       head ->rightTag = 0;
       head->leftTag = 0;
       head->data = num;
       printf("input %d left\n",num);
       head->left = createTree(head->left);
       printf("input %d right\n",num);
       head->right = createTree(head->right);
     }
   }
   return head;
}

在线索二叉树中，被线索化的节点其空闲的右指针指向后继节点，空闲的左指针指向前驱节点**

在线索化的过程中我们需要引入一个中间指针（prev）来保存某个节点，以便去寻找它的后继节点(pCur)，然后把他们连起来

/*线索化二叉树*/
node *prev; // 上次访问的节点(前驱节点）
node * frontOrderTrailTree(node *head){
   if(head == NULL) return head;

   if(head->left == NULL){ //判断是否可以线索化
      head->left = prev; // 将空闲的左节点指向前驱节点
      head->leftTag = 1; // 表示这个节点的左指针已被线索化
   }

   if(prev != NULL && prev->right == NULL){
      prev->right = head; // 将空闲的右节点指向后继节点
      prev ->rightTag = 1;// 表示这个节点的右指针已被线索化
   }
   prev = head; 

// 线索化后将prev移动过来，开始寻找head的下一个后继节点
   if(head->leftTag == 0){ // 判断当前节点的左节点有没有线索化过
     head->left = frontOrderTrailTree(head->left);// 搜寻左子树
   }
   if(head->rightTag == 0){
     head->right = frontOrderTrailTree(head->right);// 搜寻右子树
   }
   return head;
}

/* 线索二叉树的前序遍历
  如果leftTag为0说明该节点存在左子树,没有被线索化那么就沿着左孩子向下搜寻,当leftTag不再为0时说明已经到达了最底端则沿着该节点的右节点遍历其后继节点（被线索化的节点的右孩子指向后继节点）
 */
void preOrderTrailTree(node *head){
   node* cur = head;
   /*不断沿着节点往下遍历，直到所有节点全部被遍历*
   \* 线索化以后，树的最后一个节点的右孩子（后继节点）为NULL*
   */
   while (cur!=NULL)
   {
     while (cur->leftTag == 0)
     {
       printf("%d ",cur->data);
       cur = cur->left; // 沿着左孩子向下遍历直到底部
     }
     printf("%d ",cur->data);
     cur = cur->right; // 沿着右孩子寻找后继节点
   }
}

#include <stdio.h>
#include <malloc.h>

typedef struct node{
    int data;
    struct node* left,*right;
    int leftTag,rightTag;
}node;

/*创建二叉树*/
node *createTree(node *head){
    if(head == NULL){
        int num = 0;
        scanf("%d",&num);
        if(num != -1){
            head =(node*) malloc(sizeof(node));
            head->right =NULL;
            head->left = NULL;
            head ->rightTag = 0;
            head->leftTag = 0;
            head->data = num;
            printf("input %d left\n",num);
            head->left = createTree(head->left);
            printf("input %d right\n",num);
            head->right = createTree(head->right);
        }
    }
    return head;
}


/* 递归遍历创建好的二叉树
 * （用于验证线索化后的结果是否正确）*/
void frontOrderTree(node *head) {
    if (head != NULL){
        printf("%d ",head->data);
        frontOrderTree(head->left);
        frontOrderTree(head->right);
    }
}

/*线索化二叉树*/
node *prev; // 上次访问的节点(前驱节点）
node * frontOrderTrailTree(node *head){
    if(head == NULL) return head;

    if(head->left == NULL){ // 遵循左前右后
        head->left = prev; // 将空闲的左节点指向前驱节点
        head->leftTag = 1; // 表示这个节点的左指针已被线索化
    }
    if(prev != NULL && prev->right == NULL){
        prev->right = head; // 将空闲的右节点指向后继节点
        prev ->rightTag = 1;// 表示这个节点的右指针已被线索化
    }

    prev = head; // 线索化后将prev移动过来，开始寻找head的下一个后继节点
    if(head->leftTag == 0){ // 判断当前节点的左节点有没有线索化过
        head->left = frontOrderTrailTree(head->left);// 搜寻左子树
    }
    if(head->rightTag == 0){
        head->right = frontOrderTrailTree(head->right); // 搜寻右子树
    }
    return head;
}


/*线索二叉树的前序遍历
 *前序遍历左子树，如果leftTag为0说明该节点存在左子树,没有被线索化*
 *那么就沿着左孩子向下搜寻，当leftTag不再为0时说明已经到达了最底端，*
 *则沿着该节点的右节点遍历其后继节点（被线索化的节点的右孩子指向后继节点）*
 */
void preOrderTrailTree(node *head){
    node* cur = head;
    /*不断沿着节点往下遍历，直到所有节点全部被遍历*
     * 线索化以后，树的最后一个节点的右孩子（后继节点）为NULL*
     */
    while (cur!=NULL)
    {
        while (cur->leftTag == 0)
        {
            printf("%d ",cur->data);
            cur = cur->left; // 沿着左孩子向下遍历直到底部
        }
        printf("%d ",cur->data);
        cur = cur->right; // 沿着右孩子寻找后继节点
    }
}

int main(){
    node *head = NULL;
    head = createTree(head);
    printf("未线索化前序遍历:\n");
    frontOrderTree(head);
    head = frontOrderTrailTree(head);
    printf("\n线索化前序遍历:\n");
    preOrderTrailTree(head);
};
