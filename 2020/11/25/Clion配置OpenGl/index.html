Clion配置OpenGL

发表于 2020-11-25
678- 3m-
c++OpenGL环境配置

最近要使用OpenGL写课设，在配置的过程中遇到不少坑，写篇博客纪念一下在途中遇到的坑，这里我使用的是openGL的GLUT图形库

Clion下配置OpenGL

  1. 下载freeglut，注意要下载MinGW版本（因为我的clion使用的是Mingw64环境）

​ https://www.transmissionzero.co.uk/files/software/development/GLUT/freeglut-MinGW.zip

​ 2. 将下载的zip文件包解压，可以看到解压后的文件夹里还有三个文件夹。

  1. 复制include下的文件夹GL到你的/mingw64/include（如果是32位电脑则是/mingw32/include）

    上图就是已经把 GL文件夹复制到了我本地的mingw64中

  2. 复制 lib下的文件到你的/mingw64/lib,如果是64位电脑就复制x64下的.a文件，是32位的就复制外面的.a文件

​ 如图，如果你是64位电脑就复制x64文件夹下的.a文件到你自己本地的mingw64/lib文件中，32位就是mingw32/lib

  1. 复制 bin下的.dll文件（同上，64位电脑复制x64文件中的，32位电脑复制外面的）到C:\Windows\SysWOW64\目录下。如果是32位系统就是C:\Windows\SysWOW32\目录下

  2. 新建Clion工程，修改Cmake文件（非常重要）

    cmake_minimum_required(VERSION 3.17)
project(class_protect)
link_directories("C:/RT/msys64/mingw64/lib") #你的mingw64/lib的路径
set(CMAKE_CXX_STANDARD 14)
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY bin)
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -std=c++11 -D FREEGLUT_STATIC") # 重要一步，必不可少
add_executable(class_protect main.cpp) # class_protect 是我的项目名，记得换成你自己的，main.cpp是我的主函数文件
target_link_libraries(class_protect freeglut_static opengl32 winmm gdi32 glu32.lib) # 记得将最前面的项目名改程自己的
    /*测试代码*/
#include <stdio.h>
#include <GL/glut.h>
void ChangeSize(GLsizei w,GLsizei h){
 if(w<1) w=1;
 if(h<1) h=1;
 glViewport(0,0,w,h);
 glMatrixMode(GL_PROJECTION);
 glLoadIdentity();
 if(w<=h)
     glOrtho(-30.0,30.0,-30.0*(GLfloat)h/(GLfloat)w,30.0*(GLfloat)h/(GLfloat)w,-50.0,50.0);
 else
     glOrtho(-30.0*(GLfloat)w/(GLfloat)h,30.0*(GLfloat)w/(GLfloat)h,-30.0,30.0,-50.0,50.0);
 glMatrixMode(GL_MODELVIEW);
 glLoadIdentity();
 gluLookAt(0.0,0.0,5.0, 0.0,0.0,-1.0,0.0f,1.0f,0.0f);
}
void DrawMyObjects() {
 //画四边形
 glBegin(GL_QUADS);
 glVertex2f(-30,-30);
 glVertex2f(-55,-10);
 glVertex2f(-10,-5);
 glVertex2f(-10,-20);
 glColor3f(0.5,0.7,0.2);
 glEnd();
}
void RenderScene()
{
 glClear(GL_COLOR_BUFFER_BIT);
 glColor3f(1.0,1.0,0.0);
 DrawMyObjects();
 glFlush();
}
int main(int argc, char *argv[]){
 glutInit(&argc, argv);
 glutInitDisplayMode(GLUT_SINGLE|GLUT_RGB);
 glutInitWindowSize(600,600);

 glutCreateWindow("My first");

 glClearColor(0.7,0.4,0.9,0.5);

 glutReshapeFunc(ChangeSize);
 glutDisplayFunc(RenderScene);

 glutMainLoop();
 return 0;
}

    遇到的问题以及解决方案

如果出现undefined to xxxxxx的问题，先检查gcc环境配置是否正确，再检查cmake文件配置是否有问题，然后查看下载的freeglut版本与mingw版本是否一致。一在其它配置都正确的情况下最大的可能是cmake文件配置有问题。
