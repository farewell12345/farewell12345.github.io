1.栈

栈是限定仅在表尾进行插入或者删除操作的线性表，表尾称为栈顶，表头称为栈头。

特点与性质

  • LIFO（后进先出）
  • 线性数据结构，存储。查找，插入，删除操作只能访问栈的一端（栈顶）
  • 栈顶与栈底 ：允许元素插入与删除的一端称为栈顶，另一端称为栈底。
  • 进栈：向栈插入元素的操作，叫做进栈，也称压栈、入栈
  • 出栈：栈顶元素的删除操作，也叫做出栈。

利用栈进行表达式求值

中缀表达式

一般所用的表达式是将运算符号放在两运算对象的中间，比如a+b,c/d等，把这种表达式称为中缀表达式。

算数四则运算：

1.先乘除，后加减；

2.从左到右进行计算

3.先括号内，后括号外，多层括号，由内向外计算

后置表达式（逆波兰表达式）

后序表达式是将运算符放在操作数的后面

例如中缀表达式a+b转换为后置表达式就是 ab+

后置表达式的计算方式非常方遍，计算时只需要遍历表达式将表达式中的数值压入运算栈中，然后当遇到运算符时从运算栈中弹出两个数值与运算符进行运算即可
因此在使用栈对表达式求值时，我们一般先将输入的 中缀表达式转换为后缀表达式再进行运算

中置表达式转后缀表达式：

string ChangePostPositional(string c)
{
    string changed;
    //存放后置表达式的字符串
    Stack *temPopCharerator = InitStack();
    //初始化运算符栈
    for (int i = 0; i < c.size(); i++)
    {
        while (1)
        {
            if (!is_Dight(c[i]))
                //判断是不是连续数字，比如25，确保大数字被完整保留下来
            {
                changed += ',';
                //加个逗号用于分割数字和运算符
                break;
            }
            changed += c[i++];
        };
        if (c[i] == '(')//如果遇到左括号将其压入栈中，然后继续遍历
        {
            PushChar(temPopCharerator, c[i]);
        }
        else if (c[i] == ')')
            //遇到右括号时将之前压入的运算符全部弹出，直到遇到左括号停止，保证括号运算
        {
            do
            {
                char a = PopChar(temPopCharerator);
                if (a == '(')
                {
                    break;
                }
                changed += a;
            } while (1);
        }
        else
        {
            /*进行运算优先级判断，
            如果栈顶运算符优先级大于当前运算符，
            则弹出栈顶运算符，
            将当前运算符入栈，
            保证优先级大的在前
            反之直接入栈
            */
            switch (Precede(ReadCharTop(temPopCharerator), c[i]))
            {
            case '>':
                changed += PopChar(temPopCharerator);
                PushChar(temPopCharerator, c[i]);
                break;
            case '=':
            case '<':
                PushChar(temPopCharerator, c[i]);
                break;
            default:
                break;
            }
        }
    }
    do
    //结束遍历将剩下的运算符全部弹出
    {
        char a=PopChar(temPopCharerator);
        if (a==0)
        {
            continue;
        }
        changed += a;
    } while (temPopCharerator->CharTop > 0);
    return changed;
}

后缀表达式的计算：

int CalculateExpression(string c)
{
    /*进行后缀表达式运算*/
    Stack *NumStack = InitStack();
    //初始化数字栈
    Stack *operatorStack = InitStack();
    //初始化运算符栈
    for (int i = 0; i < c.size(); i++)
    {
        if (c[i] == ',')//去掉用于分隔的逗号
        {
            continue;
        }
        if (is_Dight(c[i]))
        {
            int num = 0;
            do
            {
                    num *= 10;
                    num += c[i] - '0';
                //将纯数字字符串转成Int类型方遍计算
            } while (is_Dight(c[++i]));
            PushNum(NumStack, num);
            //将转换后的数字压入数字栈
        }
        else
        {
            //遇到运算符，从数字栈里弹出两个成员与运算符进行运算并将结果重新压入数字栈
            PushNum(NumStack, Calculate(PopNum(NumStack), c[i], PopNum(NumStack)));
        }
    }
    return ReadNumTop(NumStack);
    //将运算栈的顶部元素返回，这个元素对应的值即为后置运算符的计算结果
}

2.队列

队列是线性表的一种，是一种 先进先出，后进后出 的线性表。队列限制在表的一端进行插入而另一端进行删除操作。

能够插入元素的一端叫做 队尾，能够删除的一端叫做 队首。队列简称 FIFO表

顺序队列

#include<iostream>
using namespace std;
struct Node//结点结构体（面向对象里可以看作对象
{
    string data;
    Node *Next;
};
struct LinkQue//用于充当头指针和尾指针
{
    Node *top,*tear;
};
LinkQue *IniteQue(string a)
{
    LinkQue *p=new LinkQue;//初始化一个指针结构体
    Node *Q=new Node();//初始化第一个节点
    Q->data=a;//数据入列
    Q->Next=NULL;
    p->top=Q;//固定指针到节点
    p->tear=Q;
    return p;
}
void InsertQue(LinkQue *p,string n)
{
    cout<<"输入队列"<<n<<endl;
    Node *Q=new Node;//创建新节点
    Q->data=n;//数据入列
    Q->Next=NULL;
    p->tear->Next=Q;//连接节点
    p->tear=Q;//移动底部指针
}
string ExitQue(LinkQue *p)
{
    string a= p->top->data;//拿到弹出队列的数据
    Node *N=p->top;//用指针把要弹出的节点拿出来
    p->top=p->top->Next;//移动顶部指针到下一个位置
    N->Next=NULL;//断开节点（弹出队列）
    delete(N);//删除内存
    return a;//返回弹出的数据
}
int main(int argc, char const *argv[])
{
    LinkQue *s=IniteQue(to_string(0));
    for (int i = 0; i < 50; i++)
    {
        InsertQue(s,to_string(i));
        /* code */
    }
    for (int i = 0; i < 51; i++)
    {
        cout<<"弹出队列："<<ExitQue(s)<<endl;
    }
    return 0;
}
